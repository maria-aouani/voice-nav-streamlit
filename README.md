# voice-nav-streamlit
Voice navigation app built with streamlit for blind people.

git clone https://github.com/maria-aouani/voice-nav-streamlit.git
cd voice-nav-streamlit
python -m venv .venv
# Activate virtualenv
# Windows
.venv\Scripts\activate
# Linux / Mac
source .venv/bin/activate
pip install -r requirements.txt
streamlit run main.py
